% horizontal line at height y
\newcommand{\hlineat}[2][]{%
  \pgfmathsetmacro{\y}{#2}%
  \pgfmathsetmacro{\x}{\A*(1-\y/\H)}%
  \draw[#1] (-\x,\y) -- (\x,\y);%
}

% outer lines
\coordinate (L) at (-\A,0);
\coordinate (R) at ( \A,0);
\coordinate (T) at (0,\H);
\draw[ultra thick] (L) -- (R) -- (T) -- cycle;

% inner lines
\foreach \k in {1,...,9}{
  \pgfmathsetmacro{\y}{\k*\u}
  \ifnum\k=3
    \hlineat[ultra thick]{\y}
  \else\ifnum\k=8
    \hlineat[ultra thick]{\y}
  \else
    \hlineat[thin]{\y}
  \fi\fi
}

% label between y_low und y_high
\newcommand{\layerlabelY}[3]{%

}
