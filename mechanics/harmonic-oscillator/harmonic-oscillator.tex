\documentclass[crop,tikz]{standalone}

\usepackage{pgfplots}
\usepackage{siunitx}

  inverted/.style = {
    every axis legend/.append style={
      draw=white,

      text=white
    }
  },
  every non boxed x axis/.append style={
    axis line style={-latex}
  },
  every non boxed y axis/.append style={

  }

\begin{document}
\begin{tikzpicture}
\begin{axis}[

  domain={0}:{6*pi},
  samples=100,
  smooth,
  axis y line=middle,
  axis x line=middle,
  xlabel={$t/\si{\s}$},
  ylabel={$x/\si{\m}$},
  xlabel style={right},
  ylabel style={above},
  xmin=0, xmax={6.3*pi},
  ymin=-2.2, ymax=2.4,

  legend style={at={(1,0)},anchor=north east}
  ]

  \addlegendentry{$x(t)=\SI{2}{\m}\sin(\SI{1}{\per\s} t)$};
