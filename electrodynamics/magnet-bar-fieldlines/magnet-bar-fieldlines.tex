% Plots field lines of a bar magnet of length h using the expressions
% from Martín-Luna et.al., "On the Magnetic Field of a Finite
% Solenoid", IEEE TRANSACTIONS ON MAGNETICS, VOL. 59, NO. 4, APRIL
% 2023

% Note that the expressions are written in terms of the dimensionless
% ratio r/r_0. Thus, the x-axis shows the ratio x/r_0 and the y-axis
% shows the ratio y/r_0. I.e. the borders of the solenoid are at x=-1
% and x=1.

\usepackage{pgfplots}
\usepackage{luacode}

\pgfplotsset{
  compat=1.18,
  inverted/.style = {
    every axis legend/.append style={
      draw=white,
      fill=hardblack,
      text=white
    }
  },
}

\begin{luacode*}
Ell = Ell or {}

-- 16-pt Gauss-Legendre nodes/weights on [-1,1], reduced to 8 positive nodes
Ell.xi = {
  0.9894009349916499,
  0.9445750230732326,
  0.8656312023878317,
  0.7554044083550030,
  0.6178762444026437,
  0.4580167776572274,
  0.2816035507792589,
  0.09501250983763744
}
Ell.wi = {
  0.027152459411754095,
  0.062253523938647893,
  0.095158511682492785,
  0.12462897125553387,
  0.14959598881657673,
  0.16915651939500254,
  0.18260341504492359,
  0.18945061045506850
}

function Ell.clamp(x)
  local eps = 1e-15
  if x < eps then
    return eps
  else
    return x
  end
end

function Ell.K(k)
  local kk = k*k
  local sum = 0.0
  for i=1,#Ell.xi do
    local th1 = math.pi/4*(1 + Ell.xi[i])
    local th2 = math.pi/4*(1 - Ell.xi[i])
    local s1  = math.sin(th1); s1 = s1*s1
    local s2  = math.sin(th2); s2 = s2*s2
    local r1  = Ell.clamp(1.0 - kk*s1)
    local r2  = Ell.clamp(1.0 - kk*s2)
    sum = sum + Ell.wi[i]*(1.0/math.sqrt(r1) + 1.0/math.sqrt(r2))
  end
  return math.pi/4*sum
end

function Ell.E(k)
  local kk = k*k
  local sum = 0.0
  for i=1,#Ell.xi do
    local th1 = math.pi/4*(1 + Ell.xi[i])
    local th2 = math.pi/4*(1 - Ell.xi[i])
    local s1  = math.sin(th1); s1 = s1*s1
    local s2  = math.sin(th2); s2 = s2*s2
    local r1  = Ell.clamp(1.0 - kk*s1)
    local r2  = Ell.clamp(1.0 - kk*s2)
    sum = sum + Ell.wi[i]*(math.sqrt(r1) + math.sqrt(r2))
  end
  return math.pi/4*sum
end

function Ell.Pi(n,k)
  local kk = k*k
  local sum = 0.0
  for i=1,#Ell.xi do
    local th1 = math.pi/4*(1 + Ell.xi[i])
    local th2 = math.pi/4*(1 - Ell.xi[i])
    local s1  = math.sin(th1); s1 = s1*s1
    local s2  = math.sin(th2); s2 = s2*s2
    local r1  = Ell.clamp(1.0 - kk*s1)
    local r2  = Ell.clamp(1.0 - kk*s2)
    local d1  = Ell.clamp(1.0 - n*s1)
    local d2  = Ell.clamp(1.0 - n*s2)
    sum = sum + Ell.wi[i]*(1.0/(d1*math.sqrt(r1)) + 1.0/(d2*math.sqrt(r2)))
  end
  return math.pi/4*sum
end

-- k_pm
function Ell.k(r,z,h,s)
  return math.sqrt(4*r/((1 + r)^2 + (z + 0.5*s*h)^2))
end

-- k_+
function Ell.kp(r,z,h)
  return Ell.k(r,z,h,1)
end

-- k_-
function Ell.km(r,z,h)
  return Ell.k(r,z,h,-1)
end

function sigma(r)
  return 4*r/(1 + r)^2
end

-- normalization
Ell.BO = 4*math.pi

-- I_pm
function Ell.Ipm(r,z,k)
  return -4/math.sqrt(r)*(Ell.E(k)/k - (1 - 0.5*k^2)*Ell.K(k)/k)
end

-- I'_pm
function Ell.Jpm(r,z,k,h,s)
  return k/math.sqrt(r)*(z + 0.5*s*h)*((1 - r)/(1 + r)*Ell.Pi(sigma(r),k) + Ell.K(k))
end

-- I_+
function Ell.Ip(r,z,h)
  return Ell.Ipm(r,z,Ell.kp(r,z,h),h)
end

-- I_-
function Ell.Im(r,z,h)
  return Ell.Ipm(r,z,Ell.km(r,z,h),h)
end

-- I'_+
function Ell.Jp(r,z,h)
  return Ell.Jpm(r,z,Ell.kp(r,z,h),h,1)
end

-- I'_-
function Ell.Jm(r,z,h)
  return Ell.Jpm(r,z,Ell.km(r,z,h),h,-1)
end

-- B_z(0,z)
function Ell.Bz0(r,z,h)
  return 0.5*Ell.BO*((z + 0.5*h)/math.sqrt(1 + (z + 0.5*h)^2) - (z - 0.5*h)/math.sqrt(1 + (z - 0.5*h)^2))
end

-- B_r(r,z)
function Ell.Br(r,z,h)
  if r == 0 then
    return 0
  else
    return Ell.BO/(4*math.pi)*(Ell.Im(r,z,h) - Ell.Ip(r,z,h))
  end
end

-- B_z(r,z)
function Ell.Brz(r,z,h)
  if r == 0 then
    return Ell.Bz0(r,z,h)
  else
    return Ell.BO/(4*math.pi)*(Ell.Jp(r,z,h) - Ell.Jm(r,z,h))
  end
end

-- r(x,y)
function Ell.r(x,y)
  return math.sqrt(x^2 + y^2)
end

-- B_x(x,y,z)
function Ell.Bx(x,y,z,h)
  if Ell.r(x,y) == 0 then
    return 0
  else
    return Ell.Br(Ell.r(x,y),z,h)*math.cos(math.atan2(y,x))
  end
end

-- B_y(x,y,z)
function Ell.By(x,y,z,h)
  if Ell.r(x,y) == 0 then
    return 0
  else
    return Ell.Br(Ell.r(x,y),z,h)*math.sin(math.atan2(y,x))
  end
end

-- B_z(x,y,z)
function Ell.Bz(x,y,z,h)
  return Ell.Brz(Ell.r(x,y),z,h)
end

-- |vec{B}(x,y,z)|
function Ell.Ba(x,y,z,h)
  return math.sqrt(Ell.Br(Ell.r(x,y),z,h)^2 + Ell.Bz(x,y,z,h)^2)
end

-- boundaries
xmin,xmax,ymin,ymax = -3,3,-5,5

-- Runge-Kutta integration
local function rk4_step(x,y,h,d)
  local function Fn(x,y)
    local ux,uy = Ell.Bx(x,0,y,h),Ell.Bz(x,0,y,h)
    local m = math.sqrt(ux^2 + uy^2)
    if m==0 or m~=m then return 0,0 end
    return ux/m, uy/m
  end
  local k1x,k1y = Fn(x,y)
  local k2x,k2y = Fn(x+0.5*d*k1x, y+0.5*d*k1y)
  local k3x,k3y = Fn(x+0.5*d*k2x, y+0.5*d*k2y)
  local k4x,k4y = Fn(x+d*k3x, y+d*k3y)
  return x + d/6*(k1x+2*k2x+2*k3x+k4x),
         y + d/6*(k1y+2*k2y+2*k3y+k4y)
end

-- create fieldline
function streamline(x0,y0,h,d,n)
  local step = math.abs(d)
  local eps  = step/2
  local eps2 = eps*eps
  local minsteps = 100

  local function integrate(h,dd)
    local pts = {}
    local x,y = x0,y0
    local d2  = 0.0
    local closed = false

    for i=1,n do
      pts[#pts+1] = {x,y}

      local xn,yn = rk4_step(x,y,h,dd)

      local dx,dy   = x  - x0, y  - y0
      local dxn,dyn = xn - x0, yn - y0
      local d2n = dxn*dxn + dyn*dyn
      d2 = dx*dx + dy*dy

      if i > minsteps then
        if (d2 > eps2) and (d2n <= eps2) then
          -- interpolate intersection
          local d  = math.sqrt(d2)
          local dn = math.sqrt(d2n)
          local t = (d - eps) / (d - dn)
          local xc = x + t*(xn - x)
          local yc = y + t*(yn - y)
          pts[#pts+1] = {xc,yc}
          closed = true
          break
        end
        if d2n <= eps2 then
          closed = true
          break
        end
      end

      x,y = xn,yn
      if x<xmin or x>xmax or y<ymin or y>ymax then break end
    end
    return pts, closed
  end

  -- 1) forward integration
  local fwd, closed = integrate(h,step)

  if closed then
    -- closed field line; print only once
    for i=1,#fwd do
      tex.sprint(string.format("(%.12g,%.12g)", fwd[i][1], fwd[i][2]))
    end
    return
  end

  -- 2) open field line: integrate backwards and forwards and connect them
  local bwd, _ = integrate(h,-step)

  -- integrate backwards in opposite direction (don't starting point twice)
  for i=#bwd,2,-1 do
    tex.sprint(string.format("(%.12g,%.12g)", bwd[i][1], bwd[i][2]))
  end
  for i=1,#fwd do
    tex.sprint(string.format("(%.12g,%.12g)", fwd[i][1], fwd[i][2]))
  end
end
\end{luacode*}
