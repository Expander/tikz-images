\begin{document}
\begin{tikzpicture}[inverted,inverted]
  \begin{axis}[inverted,

    domain=-5:5,

    ]
    \addplot3[
       restrict z to domain* = -3:0,
       surf,
       samples=70,
    ]{ f(x,y) };
    \node[white] at (axis cs: 3, 5, 0) { $\phi(\vec{r})$ };
    \coordinate (O) at (axis cs: -3, -5.5, 0); % origin
